!! test
Lists: loop control (break/continue)
!! input
<wikiscript>
a = [ 1, 2, 3, 4, 5, 6, 7, 8, 9 ];
foreach( e in a ) {
	if( e >= 6 & e < 9 )
		continue;
	out( e );
}
foreach( e in a ) {
	if( e == 3 )
		break;
	out( e );
}
</wikiscript>
!! result
<p>12345912
</p>
!! end
